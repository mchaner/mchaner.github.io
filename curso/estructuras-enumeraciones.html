# Que son las estructuras

Son tipos de datos definidos por el programador, estos tipos de datos contienen otros tipos de datos definidos por el lenguaje a los cuales llamaremos campos.

Si tomamos como ejemplo una agenda, la estructura se llamaría agenda:

Si llevamos esto a código quedaría de la siguiente manera:

//esto debe estar fuera del main
public struct Agenda
{
    public string Nombre;
    public string Direccion;
    public string Telefono;
    public int Edad;
}

Cuando tenemos una plantilla para mostrar una estructura podemos sobre escribir el metodo ToString(), y mostrar lo que deseamos.

using System;
using System.Text;

namespace agenda_telefonica
{
    class Program
    {
        static void Main(string[] args)
        {
            Agenda agenda;
            agenda.Nombre = "Luis";
            agenda.Direccion = "Avenida Siempreviva 742";
            agenda.Telefono = "+56-9-9856-1525";
            agenda.Edad = 26;

            Console.WriteLine(agenda.ToString());

            Console.ReadKey();
        }        
    }

    public struct Agenda
    {
        public string Nombre;
        public string Direccion;
        public string Telefono;
        public int Edad;

        public override string ToString()
        {
            StringBuilder stringBuilder = new StringBuilder();
            stringBuilder.AppendFormat("Nombre: {0}, Direccion: {1}, Telefono: {2}, Edad: {3}", Nombre, Direccion, Telefono, Edad);

            return stringBuilder.ToString();
        }
    }
}

Cuanto tenemos una estructura con múltiples campos, es conveniente crear un constructor y utilizar el objeto como instancia para inicializarlo.

using System;
using System.Text;

namespace agenda_telefonica
{
    class Program
    {
        static void Main(string[] args)
        {            
            Agenda contactos = new Agenda("Luis", "Avenida Siempreviva 742", "+56-9-9856-1525", 26);

            Console.WriteLine(contactos.ToString());

            Console.ReadKey();
        }        
    }

    public struct Agenda
    {
        public string Nombre;
        public string Direccion;
        public string Telefono;
        public int Edad;

        public Agenda(string nombre, string direccion, string telefono, int edad)
        {
            Nombre = nombre;
            Direccion = direccion;
            Telefono = telefono;
            Edad = edad;
        }

        public override string ToString()
        {
            StringBuilder stringBuilder = new StringBuilder();
            stringBuilder.AppendFormat("Nombre: {0}, Direccion: {1}, Telefono: {2}, Edad: {3}", Nombre, Direccion, Telefono, Edad);

            return stringBuilder.ToString();
        }
    }
}

Igualmente podemos sobre cargar el constructor, para tener un mayor control de los campos en la estructura

using System;
using System.Text;

namespace agenda_telefonica
{
    class Program
    {
        static void Main(string[] args)
        {            
            Agenda contactos = new Agenda("Luis", "Avenida Siempreviva 742", "+56-9-9856-1525", 26);

            Console.WriteLine(contactos.ToString());

            Console.ReadKey();
        }        
    }

    public struct Agenda
    {
        public string Nombre;
        public string Direccion;
        public string Telefono;
        public int Edad;

        public Agenda(string nombre, string direccion, string telefono)
        {
            Nombre = nombre;
            Direccion = direccion;
            Telefono = telefono;
            Edad = 0;
        }

        public Agenda(string nombre, string direccion, string telefono, int edad)
        {
            Nombre = nombre;
            Direccion = direccion;
            Telefono = telefono;
            Edad = edad;
        }

        public override string ToString()
        {
            StringBuilder stringBuilder = new StringBuilder();
            if (Edad <= 0)
            {
                stringBuilder.AppendFormat("Nombre: {0}, Direccion: {1}, Telefono: {2}", Nombre, Direccion, Telefono);
            }
            else
            {
                stringBuilder.AppendFormat("Nombre: {0}, Direccion: {1}, Telefono: {2}, Edad: {3}", Nombre, Direccion, Telefono, Edad);
            }

            return stringBuilder.ToString();
        }
    }
}
